# 求数组中的最大值

# 出题目的

  • 考察数组API

# 知识点

  • Array.sort
  • Array.reduce
  • Math.max

# 参考代码

  <body>
    <script>
      const arr = [100, 5, 20, 3, 200, 6];
      console.log(getMaxSort(arr));
      
      //Math.max()写法
      function getMaxMath(arr) {
        return Math.max(...arr);
        return Math.max.apply(null, arr);
      }


      //sort()写法
      function getMaxSort(arr) {
        arr.sort((n1, n2) => {
          //   return n1 - n2; // 从小到大排列
          return n2 - n1; // 从大到小排列
        });
        return arr[0];
      }


      //reduce()写法
      function getMaxReduce(arr) {
        return arr.reduce((n1, n2) => {
          return n1 > n2 ? n1 : n2;
        }); // 比较相邻的两个数，大的留下作为n1，再跟后面的比较
      }
    </script>
  </body>
</html>
